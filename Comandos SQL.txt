CREATE DATABASE RJCARS;
USE RJCARS;

CREATE TABLE AUTOS (
VIN int AUTO_INCREMENT NOT NULL,
MARCA varchar(20) NOT NULL,
MODELO varchar(30) NOT NULL,
TIPO varchar(15) NOT NULL,
AÑO year NOT NULL,
PRECIO float NOT NULL,
PRIMARY KEY (VIN)
);

CREATE TABLE CLIENTES (
CEDULA varchar(15) NOT NULL,
NOMBRE varchar(25) NOT NULL,
APELLIDO varchar(25) NOT NULL,
DIRECCION varchar(100) NOT NULL,
TELEFONO varchar(15) NOT NULL,
PRIMARY KEY (CEDULA)
);

CREATE TABLE EMPLEADOS (
NoEMP int AUTO_INCREMENT NOT NULL,
CEDULA varchar(15) NOT NULL,
NOMBRE varchar(25) NOT NULL,
APELLIDO varchar(25) NOT NULL,
CARGO varchar(25) NOT NULL,
DIRECCION varchar(100) NOT NULL,
SALARIO float NOT NULL,
PRIMARY KEY (NoEMP)
);

CREATE TABLE TRANSACCIONES (
NoTRAN int AUTO_INCREMENT NOT NULL,
NoEMP int NOT NULL,
CedCLIENTE varchar(15) NOT NULL,
VIN int NOT NULL UNIQUE,
VALOR float NOT NULL,
FECHA datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (NoTRAN),
FOREIGN KEY (NoEMP) REFERENCES EMPLEADOS(NoEMP),
FOREIGN KEY (CedCLIENTE) REFERENCES CLIENTES(CEDULA),
FOREIGN KEY (VIN) REFERENCES AUTOS(VIN)
);
